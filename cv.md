function highAndLow(numbers) {
  let arrNums = numbers.split(' ');
  let maxN = arrNums[0];
  let minN = arrNums[0];
  for (let i = 1; i < arrNums.length; i++) {
    if (+arrNums[i] > maxN)
      maxN = +arrNums[i];
    if (+arrNums[i] < minN)
      minN = +arrNums[i];
  }
  return `${maxN} ${minN}`;
}
